// Install deps first: // npm init -y // npm install express const express = require("express"); const path = require("path"); const app = express(); const PORT = process.env.PORT || 3000; const ADMIN_PASSWORD = "i3u8rify47y"; // In-memory log storage (use a DB or file in production) const logs = []; // Middleware to parse JSON/form bodies app.use(express.urlencoded({ extended: true })); app.use(express.json()); // Log every visit to any route except /admin and /login app.use((req, res, next) => { if (req.path !== "/admin" && req.path !== "/login") { const ip = req.headers["x-forwarded-for"]?.split(",")[0].trim() || req.socket.remoteAddress; const ua = req.headers["user-agent"]; const time = new Date().toISOString(); logs.push({ ip, ua, time, path: req.path }); console.log("Logged visit:", { ip, ua, time, path: req.path }); } next(); }); // Simple session-like flag in memory (for demo only) let adminLoggedIn = false; // Landing page with login form app.get("/", (req, res) => { res.send(` Secure Access

Admin Login


`); }); // Handle login app.post("/login", (req, res) => { const { password } = req.body; if (password === ADMIN_PASSWORD) { adminLoggedIn = true; return res.redirect("/admin"); } return res.send(` Login Failed

Incorrect password

Go back `); }); // Admin dashboard app.get("/admin", (req, res) => { if (!adminLoggedIn) { return res.redirect("/"); } const rows = logs .map( (l) => ` ${l.ip} ${l.time} ${l.path} ${l.ua} ` ) .join(""); res.send(` Visitor Logs

Visitor Logs

${rows || ''}
IP Time Path User Agent
No logs yet.
`); }); // Logout app.post("/logout", (req, res) => { adminLoggedIn = false; res.redirect("/"); }); // Optional: JSON logs endpoint (if you ever want to fetch via JS) app.get("/logs.json", (req, res) => { if (!adminLoggedIn) return res.status(403).json({ error: "Unauthorized" }); res.json(logs); }); app.listen(PORT, () => { console.log(`Server running on http://localhost:${PORT}`); });
